## 一、注意力机制

### Self-Attention（自注意力）

**公式**：`Attention(Q, K, V) = softmax(QK^T / √d_k) V`

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class SelfAttention(nn.Module):
    def __init__(self, embed_dim):
        super(SelfAttention, self).__init__()
        self.embed_dim = embed_dim
        # 定义三个权重矩阵：W_q、W_k、W_v
        self.W_q = nn.Linear(embed_dim, embed_dim)  # 线性变换
        self.W_k = nn.Linear(embed_dim, embed_dim)
        self.W_v = nn.Linear(embed_dim, embed_dim)

    def forward(self, x, mask=None):
        """
        x:    [batch_size, seq_len, embed_dim]
        mask: [batch_size, seq_len, seq_len]  True=可见 / False=屏蔽
        """
        # 计算 Q, K, V 三个矩阵
        q = self.W_q(x)  # [batch, seq_len, embed_dim]
        k = self.W_k(x)
        v = self.W_v(x)

        # 计算 QK^T / √d_k，即向量之间的相关度
        scale = torch.sqrt(torch.tensor(float(self.embed_dim), device=x.device))
        scores = torch.matmul(q, k.transpose(-1, -2)) / scale

        # 应用 mask：将被 mask 的位置设为 -inf，softmax 后概率趋近于 0
        if mask is not None:
            scores = scores.masked_fill(~mask, float('-inf'))

        # 计算向量权重，softmax 归一化
        attn_weights = F.softmax(scores, dim=-1)
        # 计算输出
        output = torch.matmul(attn_weights, v)
        return output
```

**面试要点**：为什么要除以 √d_k？防止内积方差随维度线性增长，softmax 饱和导致梯度消失。mask 有两种——causal mask（上三角为 False）和 padding mask（忽略 [PAD]）。

### Multi-Head Attention（MHA，多头注意力）

**核心**：将 embed_dim 拆成 num_heads × head_dim，各头独立计算注意力后拼接。从代码直观理解，每个头中的计算与原本的 SA 没有区别，最后拼接再经 W_o 线性变换。

```python
import torch
import torch.nn as nn

class MultiHeadAttention(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super(MultiHeadAttention, self).__init__()
        assert embed_dim % num_heads == 0, "embed_dim must be divisible by num_heads"
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads

        self.w_q = nn.Linear(embed_dim, embed_dim)
        self.w_k = nn.Linear(embed_dim, embed_dim)
        self.w_v = nn.Linear(embed_dim, embed_dim)
        self.w_o = nn.Linear(embed_dim, embed_dim)

    def split_heads(self, x):
        """
        将最后一维拆分成 num_heads * head_dim
        x: [batch_size, seq_len, embed_dim]
        return: [batch_size, num_heads, seq_len, head_dim]
        """
        batch_size, seq_len, _ = x.shape
        x = x.view(batch_size, seq_len, self.num_heads, self.head_dim)
        return x.transpose(1, 2)  # 交换 seq_len 和 num_heads 维度

    def forward(self, hidden_states, mask=None):
        batch_size = hidden_states.size(0)

        # 线性变换得到 Q, K, V
        q = self.w_q(hidden_states)  # [batch_size, seq_len, embed_dim]
        k = self.w_k(hidden_states)
        v = self.w_v(hidden_states)

        # 多头切分
        q = self.split_heads(q)  # [batch_size, num_heads, seq_len, head_dim]
        k = self.split_heads(k)
        v = self.split_heads(v)

        # 注意力计算: softmax(QK^T / √d_k) V
        scores = torch.matmul(q, k.transpose(-2, -1)) / torch.sqrt(torch.tensor(self.head_dim, dtype=torch.float32))
        if mask is not None:
            scores = scores.masked_fill(~mask, float('-inf'))

        attention = torch.softmax(scores, dim=-1)
        output = torch.matmul(attention, v)  # [batch_size, num_heads, seq_len, head_dim]

        # 拼接多头：先交换维度，再合并最后两个维度
        output = output.transpose(1, 2).contiguous().view(batch_size, -1, self.num_heads * self.head_dim)

        # 输出线性变换
        output = self.w_o(output)
        return output
```

**维度变化回顾**：
```
q.dim == k.dim == v.dim == (bs, seq_len, hidden_dim)
    ↓ reshape + transpose
q.dim == k.dim == v.dim == (bs, num_head, seq_len, head_dim)
    ↓ matmul QK^T
(q * k^T).dim == attn_weight.dim == (bs, num_head, seq_len, seq_len)
    ↓ matmul with V
(attn_weight * v).dim == output.dim == (bs, num_head, seq_len, head_dim)
    ↓ transpose + concat
final_output.dim == (bs, seq_len, hidden_dim)
```

**面试要点**：多头的好处（多视角/多尺度/增强表达/降低单头维度/结构先验）；为什么需要 `contiguous()`（transpose 后内存不连续，view 之前需要先连续化）。

### Multi-Query Attention（MQA，多查询注意力）

**核心**：所有 Q 头共享同一对 K/V 投影矩阵。wk 和 wv 都是直接将 hidden_dim 转化为 head_dim，之后 split_head 时 k 和 v 并不分裂（在原来基础上加入 num_head=1 这个维度），后续计算时不同 head 的 q 都要和同一个 k 和 v 进行计算，因为计算时是广播。

```python
import torch
import torch.nn as nn

class MultiQuerySelfAttention(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super(MultiQuerySelfAttention, self).__init__()
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads

        self.w_q = nn.Linear(embed_dim, embed_dim)
        # MQA: K 和 V 共享，只有 1 个头，输出维度是 head_dim 而非 embed_dim
        self.w_k = nn.Linear(embed_dim, self.head_dim)
        self.w_v = nn.Linear(embed_dim, self.head_dim)
        self.w_o = nn.Linear(embed_dim, embed_dim)

    def split_heads(self, hidden, num_heads=None):
        """
        如果 num_heads=None → Q 的多头拆分
        如果 num_heads=1 → K/V 的单头拆分
        """
        batch_size, seq_len = hidden.size()[:2]
        if num_heads is None:
            num_heads = self.num_heads
        x = hidden.view(batch_size, seq_len, num_heads, self.head_dim).transpose(1, 2)
        return x

    def forward(self, hidden_states, mask=None):
        batch_size = hidden_states.size(0)

        # 线性变换
        q = self.w_q(hidden_states)  # [batch, seq_len, num_heads * head_dim]
        k = self.w_k(hidden_states)  # [batch, seq_len, 1 * head_dim]
        v = self.w_v(hidden_states)

        # 多头切分（K/V 以 head_num=1 拆分，广播到 num_heads）
        q = self.split_heads(q, self.num_heads)  # [batch, num_heads, seq, head_dim]
        k = self.split_heads(k, 1)                # [batch, 1, seq, head_dim]
        v = self.split_heads(v, 1)

        # 注意力计算：广播机制自动将 k, v 从 1 个头扩展到 num_heads 个头
        scores = torch.matmul(q, k.transpose(-2, -1)) / torch.sqrt(torch.tensor(self.head_dim, dtype=torch.float32))
        if mask is not None:
            scores = scores.masked_fill(~mask, float('-inf'))

        attention = torch.softmax(scores, dim=-1)
        output = torch.matmul(attention, v)  # [batch, num_heads, seq, head_dim]

        # 合并多头
        output = output.transpose(1, 2).contiguous().view(batch_size, -1, self.num_heads * self.head_dim)
        output = self.w_o(output)
        return output
```

**面试要点**：MQA 节省 KV 参数 + KV Cache 显存 + 计算量（约 97% KV Cache 节省）；代价是 K/V 表达能力下降；MHA → MQA 通过均值池化多个头的 K/V 投影矩阵 + 少量额外预训练。

### Grouped-Query Attention（GQA，分组查询注意力）

**核心**：Query 分组，每组共享一对 K/V。是 MHA 和 MQA 的折中——当组数为头数时为 MHA，当组数为 1 时为 MQA。

```
MHA：每组 1 个 Q → num_kv_heads = num_heads（如 32 组）
GQA：每组 n 个 Q → num_kv_heads = num_heads / n（如 8 组，每组 4 Q）
MQA：所有 Q 同组 → num_kv_heads = 1
```

```python
import torch
import torch.nn as nn

class GroupedQueryAttention(nn.Module):
    def __init__(self, embed_dim, num_heads, num_kv_heads):
        super(GroupedQueryAttention, self).__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be divisible by num_kv_heads"
        self.num_heads    = num_heads       # Q 的头数
        self.num_kv_heads = num_kv_heads     # K/V 的头数（组数）
        self.head_dim     = embed_dim // num_heads
        self.q_per_kv     = num_heads // num_kv_heads  # 每组多少个 Q 头

        self.w_q = nn.Linear(embed_dim, num_heads    * self.head_dim)
        self.w_k = nn.Linear(embed_dim, num_kv_heads * self.head_dim)
        self.w_v = nn.Linear(embed_dim, num_kv_heads * self.head_dim)
        self.w_o = nn.Linear(num_heads * self.head_dim, embed_dim)

    def split_q_heads(self, hidden):
        """Q 按 num_heads 拆分"""
        batch_size, seq_len, _ = hidden.shape
        return hidden.view(batch_size, seq_len, self.num_heads, self.head_dim).transpose(1, 2)

    def split_kv_heads(self, hidden):
        """K/V 按 num_kv_heads 拆分，再 expand 到 num_heads"""
        batch_size, seq_len, _ = hidden.shape
        hidden = hidden.view(batch_size, seq_len, self.num_kv_heads, self.head_dim)
        hidden = hidden.transpose(1, 2)  # [batch, num_kv_heads, seq, head_dim]

        # expand（更高效，不复制数据）—— 从 num_kv_heads 扩展到 num_heads
        hidden = hidden[:, :, None, :, :]  # [batch, num_kv_heads, 1, seq, head_dim]
        hidden = hidden.expand(batch_size, self.num_kv_heads, self.q_per_kv, seq_len, self.head_dim)
        hidden = hidden.reshape(batch_size, self.num_heads, seq_len, self.head_dim)
        return hidden

    def forward(self, hidden_states, mask=None):
        batch_size = hidden_states.size(0)

        # 线性变换
        q = self.w_q(hidden_states)  # [batch, seq_len, num_heads * head_dim]
        k = self.w_k(hidden_states)  # [batch, seq_len, num_kv_heads * head_dim]
        v = self.w_v(hidden_states)

        # 多头切分
        q = self.split_q_heads(q)   # [batch, num_heads, seq, head_dim]
        k = self.split_kv_heads(k)  # [batch, num_heads, seq, head_dim] ← 已 expand
        v = self.split_kv_heads(v)

        # 注意力计算
        scores = torch.matmul(q, k.transpose(-2, -1)) / torch.sqrt(torch.tensor(self.head_dim, dtype=torch.float32))
        if mask is not None:
            scores = scores.masked_fill(~mask, float('-inf'))

        attention = torch.softmax(scores, dim=-1)
        output = torch.matmul(attention, v)

        # 合并多头
        output = output.transpose(1, 2).contiguous().view(batch_size, -1, self.num_heads * self.head_dim)
        output = self.w_o(output)
        return output
```

**面试要点**：GQA 相比 MQA 更好的质量-速度 trade-off；Llama 2 70B 配置 64 Q + 8 KV（GQA-8）；expand 而非 repeat（expand 是 view 操作不实际复制数据，更高效）。

## 二、位置编码

### RoPE（旋转位置编码）

**原理**：通过二维旋转矩阵将位置信息编码到 Q、K 向量中。作用在**每一层**的 Q 和 K 上，而非仅输入 embedding。

- 组别越小（维度越低）→ 角速率 θ 越大 → 转动角度越大 → 高频 → 捕捉局部关系
- 组别越大（维度越高）→ 角速率 θ 越小 → 转动角度越小 → 低频 → 捕捉长上下文关系

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

def rope(x):
    """旋转位置编码实现"""
    batch_size, seq_len, head_dim = x.shape
    device = x.device

    # 计算 theta = 10000^{-2i/d}，频率随维度递减
    theta = 10000 ** (-torch.arange(0, head_dim, 2, device=device) / head_dim)
    # 生成位置索引
    pos = torch.arange(seq_len, device=device).unsqueeze(1)
    # 计算旋转角度 = 位置 × 频率
    angles = pos * theta  # [seq_len, head_dim/2]

    # 计算 cos 和 sin
    cos = torch.cos(angles)
    sin = torch.sin(angles)

    # 拆分输入数据，每两个维度为一组
    x1, x2 = x[..., 0::2], x[..., 1::2]

    # 应用二维旋转变换
    rotated_x1 = x1 * cos - x2 * sin
    rotated_x2 = x1 * sin + x2 * cos

    # 交替拼接回去，形成完整的旋转后向量
    return torch.stack([rotated_x1, rotated_x2], dim=-1).flatten(-2)


class MultiHeadAttentionWithRoPE(nn.Module):
    """带 RoPE 位置编码的多头注意力"""
    def __init__(self, embed_dim, num_heads):
        super().__init__()
        self.embed_dim = embed_dim
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads
        assert self.head_dim * num_heads == embed_dim, "嵌入维度必须能被头数整除"

        # 显式定义 W_q、W_k、W_v 矩阵
        self.W_q = nn.Linear(embed_dim, embed_dim)  # 查询权重矩阵
        self.W_k = nn.Linear(embed_dim, embed_dim)  # 键权重矩阵
        self.W_v = nn.Linear(embed_dim, embed_dim)  # 值权重矩阵
        self.out_proj = nn.Linear(embed_dim, embed_dim)  # 输出投影矩阵

    def forward(self, x, mask=None):
        batch_size, seq_len, embed_dim = x.shape

        # 使用显式的 W_q、W_k、W_v 计算 Q、K、V
        q = self.W_q(x)  # [batch_size, seq_len, embed_dim]
        k = self.W_k(x)
        v = self.W_v(x)

        # 重塑为多头格式: (batch_size, num_heads, seq_len, head_dim)
        q = q.view(batch_size, seq_len, self.num_heads, self.head_dim).transpose(1, 2)
        k = k.view(batch_size, seq_len, self.num_heads, self.head_dim).transpose(1, 2)
        v = v.view(batch_size, seq_len, self.num_heads, self.head_dim).transpose(1, 2)

        # 对 Q 和 K 应用 RoPE 位置编码（V 不需要位置信息）
        q_flat = q.reshape(-1, seq_len, self.head_dim)
        k_flat = k.reshape(-1, seq_len, self.head_dim)
        q = rope(q_flat).view_as(q)
        k = rope(k_flat).view_as(k)

        # 注意力计算
        scores = torch.matmul(q, k.transpose(-2, -1)) / (self.head_dim ** 0.5)
        if mask is not None:
            scores = scores.masked_fill(~mask, float('-inf'))

        attn = torch.softmax(scores, dim=-1)
        out  = torch.matmul(attn, v)
        out  = out.transpose(1, 2).contiguous().view(batch_size, seq_len, embed_dim)
        return self.out_proj(out)
```

**面试要点**：
- RoPE vs 正弦编码？RoPE 作用在每层 Q/K 上（而非仅输入 embedding），点积结果仅依赖相对位置
- 为什么 V 不需要位置编码？QK^T 已经包含位置信息，V 加权求和时只需内容信息
- 不同组别/维度的频率设计：低频捕捉长距离结构，高频捕捉局部细节

## 三、归一化

### BatchNorm1d

沿 batch 维度归一化——对一批样本中同一位置 token 的所有特征求统计量。

```python
import torch
import torch.nn as nn

class BatchNorm1d(nn.Module):
    def __init__(self, dim, eps=1e-5, momentum=0.1):
        super().__init__()
        self.eps = eps
        self.momentum = momentum
        self.gamma = nn.Parameter(torch.ones(dim))   # 可学习缩放参数
        self.beta  = nn.Parameter(torch.zeros(dim))   # 可学习平移参数
        # 运行时统计量（训练时更新，推理时使用）
        self.register_buffer("running_mean", torch.zeros(dim))
        self.register_buffer("running_var",  torch.ones(dim))

    def forward(self, x):
        """
        x: [batch, dim] 或 [batch, seq_len, dim]
        归一化沿 batch 维度
        """
        if self.training:
            # 训练模式：用当前 batch 的统计量
            if x.dim() == 2:
                mean = x.mean(dim=0)  # [dim]
                var  = x.var(dim=0, unbiased=False)  # 用 m 而非 m-1
            elif x.dim() == 3:
                mean = x.mean(dim=(0, 1))  # 对 batch 和 seq_len 同时求均值
                var  = x.var(dim=(0, 1), unbiased=False)
            else:
                raise ValueError(f"Unsupported input dimension: {x.dim()}")

            # 移动平均更新 running statistics
            self.running_mean = (1 - self.momentum) * self.running_mean + self.momentum * mean
            self.running_var  = (1 - self.momentum) * self.running_var  + self.momentum * var
        else:
            # 推理模式：用训练时累积的统计量
            mean, var = self.running_mean, self.running_var

        # 归一化 + 仿射变换
        x_norm = (x - mean) / torch.sqrt(var + self.eps)
        return self.gamma * x_norm + self.beta
```

**面试要点**：momentum 平滑更新 running statistics；unbiased=False 用 m 而非 m-1 做分母；BN 为什么不适合 NLP——变长序列、PAD 干扰、跨样本统计破坏 token 独立性。

### LayerNorm

沿 hidden 维度归一化——对每个样本中单个 token 的所有特征维度独立计算统计量。

```python
import torch
from torch import nn

class LayerNorm(nn.Module):
    def __init__(self, d_model, eps=1e-12):
        super().__init__()
        self.gamma = nn.Parameter(torch.ones(d_model))  # 可学习缩放参数
        self.beta  = nn.Parameter(torch.zeros(d_model))  # 可学习平移参数
        self.eps   = eps

    def forward(self, x):
        """
        x: [batch_size, seq_len, d_model]
        LayerNorm 沿最后一个维度（hidden 维度）做归一化
        """
        # keepdim=True 保持维度，方便广播
        mean = x.mean(dim=-1, keepdim=True)   # [batch_size, seq_len, 1]
        var  = x.var(dim=-1, unbiased=False, keepdim=True)

        out = (x - mean) / torch.sqrt(var + self.eps)
        out = self.gamma * out + self.beta  # 仿射变换
        return out
```

**面试要点**：
- keepdim=True 的作用？保持维度，方便广播运算
- LN 和 BN 的区别？归一化维度不同——LN 沿 hidden 维度，BN 沿 batch 维度
- 为什么 Transformer 用 LN 不用 BN？和 batch size 无关、适合逐 token 表示学习、推理时更稳定

### RMSNorm

不减均值，仅做尺度归一化。更快更省参数——从三次遍历（均值、标准差、更新）变为两次（均方根、更新）。

```python
import torch
import torch.nn as nn

class RMSNorm(nn.Module):
    def __init__(self, dim, eps=1e-8):
        super().__init__()
        self.eps = eps
        self.weight = nn.Parameter(torch.ones(dim))  # γ 参数（没有 β）

    def forward(self, x):
        """
        x: [batch_size, seq_len, dim]
        不减均值，只做尺度归一化
        """
        # 方法1：先算 L2 范数，再除以 sqrt(d)
        norm = x.norm(dim=-1, keepdim=True)   # L2 范数
        rms  = norm / (x.size(-1) ** 0.5)      # Root Mean Square

        # 方法2：直接计算均方根（等价）
        # rms = torch.sqrt(torch.mean(x ** 2, dim=-1, keepdim=True) + self.eps)

        x_normed = x / (rms + self.eps)
        return self.weight * x_normed
```

**面试要点**：无均值减法 + 无 β 参数 → 计算更快、更少内存访问、对 GPU/TPU 更友好；Llama/Mistral/Qwen 等主流 LLM 均采用 RMSNorm。

## 四、激活函数

```python
import numpy as np
from scipy.stats import norm

def gelu(x):
    """Gaussian Error Linear Unit：BERT/GPT 标准激活函数"""
    return x * norm.cdf(x)

def relu(x):
    """ReLU：简单高效，负半区为零（可能导致 Dead ReLU）"""
    return np.maximum(0, x)

def swish(x, beta=1):
    """Swish/SiLU：β=1 时即 Llama 使用的 SiLU"""
    return x * (1 / (1 + np.exp(-beta * x)))

def sigmoid(x):
    """输出 (0,1)，适合二分类输出层"""
    return 1 / (1 + np.exp(-x))

def tanh(x):
    """输出 (-1,1)，零中心"""
    return np.tanh(x)

# GLU 结构（不是独立激活函数，而是一种层结构）
# GLU(X)    = (XW + b) ⊙ σ(XV + c)
# SwiGLU(X) = (XW + b) ⊙ SiLU(XV + c)  ← Llama/PaLM 使用的 FFN 结构
```

**面试要点**：
- ReLU 缺点：负半区梯度为零，"死亡神经元"
- GELU 相比 ReLU：在 x=0 附近平滑过渡，提供概率性门控机制
- SwiGLU 优势：综合 GLU 门控能力 + Swish 平滑性，主流大模型标配

## 五、损失函数

### Cross-Entropy Loss（交叉熵损失）

```python
import numpy as np

def softmax(x):
    """稳定的 softmax 实现：减去最大值防止溢出"""
    x_shifted = x - np.max(x, axis=-1, keepdims=True)
    exp_x = np.exp(x_shifted)
    return exp_x / np.sum(exp_x, axis=-1, keepdims=True)

def cross_entropy_loss(logits, targets):
    """
    计算交叉熵损失函数。
    logits:  [batch_size, num_classes]  模型未归一化输出
    targets: [batch_size]               真实类别标签（整数，非 one-hot）
    """
    batch_size = logits.shape[0]

    # 1. 计算 softmax
    probs = softmax(logits)  # [batch_size, num_classes]

    # 2. 取出真实类别对应的概率（推荐用索引方式，比 one-hot 相乘更高效）
    true_probs = probs[np.arange(batch_size), targets]

    # 3. 避免 log(0) 的情况，做微小平滑处理
    eps = 1e-15
    true_probs = np.clip(true_probs, eps, 1 - eps)

    # 4. 取负对数并求平均
    loss = -np.mean(np.log(true_probs))
    return loss
```

**注意**：PyTorch 的 `nn.CrossEntropyLoss()` 内部已经做了 softmax，传入的应该是 **logits** 而非 softmax 后的概率。

**面试要点**：
- 为什么减去 max？防止 exp(x) 数值上溢
- CE vs NLLLoss？CE = log_softmax + NLLLoss
- 为什么交叉熵适合分类？直接最大化正确类的对数似然

### Focal Loss

**公式**：FL(p_t) = -α (1 - p_t)^γ log(p_t)

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class FocalLoss(nn.Module):
    """
    Focal Loss for classification
    Args:
        alpha: 平衡因子（0-1之间，默认为0.25，常用于二分类）
        gamma: 难样本调节因子（常用 2.0）
        reduction: 'mean' | 'sum' | 'none'
    """
    def __init__(self, alpha=0.25, gamma=2.0, reduction='mean'):
        super(FocalLoss, self).__init__()
        self.alpha = alpha
        self.gamma = gamma
        self.reduction = reduction

    def forward(self, logits, targets):
        """
        logits:  [N, C] 模型未归一化输出
        targets: [N] 真实类别 (0 ~ C-1)
        """
        # 1. 计算预测概率
        probs = F.softmax(logits, dim=-1)  # [N, C]

        # 2. 取出真实类别对应的概率
        pt = probs[torch.arange(len(targets)), targets]  # [N]

        # 3. focal loss 公式
        loss = -self.alpha * (1 - pt) ** self.gamma * torch.log(pt + 1e-12)

        # 4. 归约
        if self.reduction == 'mean':
            return loss.mean()
        elif self.reduction == 'sum':
            return loss.sum()
        else:
            return loss
```

**面试要点**：Focal Loss 解决样本不平衡问题（如目标检测中背景远多于目标）；γ 越大对易分类样本降权越强，聚焦难样本；α 平衡正负样本权重。

### Pairwise Loss（Reward Model）

**公式**：L = -logsigmoid(chosen - rejected - margin)

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class PairwiseLoss(nn.Module):
    """
    Pairwise Loss for Reward Model
    """
    def forward(self, chosen_reward, reject_reward, margin=0.0):
        if margin is not None:
            loss = -F.logsigmoid(chosen_reward - reject_reward - margin)
        else:
            loss = -F.logsigmoid(chosen_reward - reject_reward)
        return loss.mean()


class LogExpLoss(nn.Module):
    """
    Pairwise Loss for Reward Model（数值更稳定的版本）
    Details: https://arxiv.org/abs/2204.05862
    """
    def forward(self, chosen_reward, reject_reward, margin=0.0):
        loss = torch.log(1 + torch.exp(reject_reward - chosen_reward)).mean()
        return loss
```

**面试要点**：PairwiseLoss 自带 margin 效果——在 0 附近也会产生 loss 更新模型拉大正负例差距；为什么用 logsigmoid 而不是 sigmoid？梯度更稳定。

### GPT Language Model Loss（SFT Loss）

**核心思想**：自回归语言模型，预测下一个 token。核心操作——`logits[:, :-1, :]` 预测 `labels[:, 1:]`。

```python
import torch
import torch.nn as nn
import torch.distributed as dist

class GPTLMLoss(nn.Module):
    """
    GPT Language Model Loss
    """
    def __init__(self, ring_attn_group=None):
        super().__init__()
        self.IGNORE_INDEX = -100
        self.loss = nn.CrossEntropyLoss(ignore_index=self.IGNORE_INDEX)

        self.ring_attn_group = ring_attn_group
        if self.ring_attn_group:
            self.ring_attn_rank = dist.get_rank(self.ring_attn_group)
            self.ring_attn_world_size = dist.get_world_size(self.ring_attn_group)

    def forward(self, logits: torch.Tensor, labels: torch.Tensor) -> torch.Tensor:
        # RingAttention 处理
        if self.ring_attn_group is not None:
            total_seq_len = labels.size(-1)
            seq_len_per_process = total_seq_len // self.ring_attn_world_size
            start_idx = self.ring_attn_rank * seq_len_per_process
            end_idx = min(start_idx + seq_len_per_process, total_seq_len)
            labels = labels[..., start_idx:end_idx]

            shift_logits = logits[..., :-1, :].contiguous()
            shift_labels = labels[..., 1:].contiguous()

            # if labels are all IGNORE_INDEX, then nn.CrossEntropyLoss will be nan
            if torch.all(shift_labels == self.IGNORE_INDEX):
                # Use mean of logits multiplied by 0 to maintain gradient flow
                loss = shift_logits.mean() * 0
            else:
                loss = self.loss(shift_logits.view(-1, shift_logits.size(-1)),
                                 shift_labels.view(-1))

            dist.all_reduce(loss, op=dist.ReduceOp.SUM, group=self.ring_attn_group)
            loss = loss / self.ring_attn_world_size
        else:
            # 普通 SFT Loss
            shift_logits = logits[..., :-1, :].contiguous()
            shift_labels = labels[..., 1:].contiguous()
            loss = self.loss(shift_logits.view(-1, shift_logits.size(-1)),
                             shift_labels.view(-1))

        return loss
```

**面试要点**：
- 为什么要 shift？自回归：当前 token 预测下一个 token
- ignore_index=-100 的作用？忽略 [PAD] token 位置的 loss
- SFT 时通常把 prompt 部分的 loss mask 掉，只学习回答部分

## 六、KL 散度估计

```python
import torch
import torch.distributions as dis

def compute_kl_estimators(p_log_probs, q_log_probs):
    """
    从分布 q 中采样，计算 D_KL(q || p) 的三种估计量
    p_log_probs: [N]  log p(x)
    q_log_probs: [N]  log q(x)，其中 x ~ q
    """
    log_r = p_log_probs - q_log_probs   # log(p(x)/q(x))

    # k1: 按定义直接估计，无偏但方差大（log 可取负值）
    k1 = -log_r

    # k2: f-散度二阶近似，有偏但方差小（恒正）
    k2 = (log_r ** 2) / 2

    # k3: Control variate 方法，无偏且方差小
    # D_KL(q || p) = E_q [p/q - 1 - log(p/q)] = E_q [exp(-log_r) - 1 + log_r]
    k3 = (log_r.exp() - 1) - log_r

    return k1.mean(), k2.mean(), k3.mean()


def validate_kl_estimators():
    """验证三种估计量的偏差和方差"""
    # 构造两个正态分布
    p = dis.Normal(loc=0, scale=1)
    q = dis.Normal(loc=0.1, scale=1)

    # 真实的 KL 散度
    true_kl = dis.kl_divergence(q, p)  # D_KL(q || p)
    print(f"True KL: {true_kl.item():.4f}")

    # 采样 N 个点
    N = 10_000_000
    x = q.sample(sample_shape=(N,))  # 从 q 中采样

    # 计算 log_prob
    p_log_probs = p.log_prob(x)
    q_log_probs = q.log_prob(x)

    # 计算三种估计量
    k1, k2, k3 = compute_kl_estimators(p_log_probs, q_log_probs)

    print(f"\n各种估计量：")
    print(f"k1 (unbiased, high var): {k1.item():.4f}")
    print(f"k2 (biased, low var):   {k2.item():.4f}")
    print(f"k3 (unbiased, low var):  {k3.item():.4f}")

    # 计算偏差
    print(f"\n偏差：")
    print(f"k1 bias: {abs(k1 - true_kl).item():.6f}")
    print(f"k2 bias: {abs(k2 - true_kl).item():.6f}")
    print(f"k3 bias: {abs(k3 - true_kl).item():.6f}")
```

| 估计量 | 公式 | 偏差 | 方差 | 说明 |
|-------|------|-----|------|------|
| k1 | -log r | 无偏 | 高 | 按定义直接估计，log 可取负值 |
| k2 | (log r)² / 2 | 有偏 | 低 | f-散度二阶近似，恒正 |
| k3 | (r - 1) - log r | 无偏 | 低 | Control variate，GRPO 常用 |

> (标准差/方差) 越大，代表分布离平均值越远，曲线整体看起来更平坦
> (标准差/方差) 越小，代表分布越靠近均值，即均值附近的概率更大，曲线整体看起来更尖锐

**面试要点**：为什么需要估计？词表 200k+ token，精确计算 KL 代价过大。k3 为何无偏？基于 control variate 技术，加上期望为 0 的项 (r-1)。实际中推荐 k3——GRPO 等大模型 RL 的常用选择。

## 七、强化学习相关实现

### On-Policy vs Off-Policy 直观理解

```python
# === On-Policy 类比 ===
# 生成数据和训练都使用同一个模型（策略）
#
# 老师批评和表扬都是根据小明的言行 policy，
# 而小明的言行 policy 也是根据老师的批评改正。
# 老师根据小明的行为 policy 玩手机，产生批评（生成数据）
# 小明根据老师的行为减小玩手机（训练模型）
#
# → 每份数据只能学一次，学完必须重新采样

# === Off-Policy 类比 ===
# off policy：择其善者而从之，其不善者而改之
#
# 根据老师对小明的批评，如果其他人自己也玩手机的话：
# 如果自己玩的更多 → 需要修改的也更多一些
# 如果自己玩的更少 → 需要修改的也会少一些
#
# → 利用重要性采样复用旧数据

# === 重要性采样核心公式 ===
# E_{x~p_old} [f(x) * p_new(x)/p_old(x)] = E_{x~p_new} [f(x)]
# ratio = π_θ(a|s) / π_ref(a|s)  就是重要性权重
```

### GAE（广义优势估计）

**公式**：δ_t = r_t + γ V(s_{t+1}) - V(s_t)，Â_t = Σ (γλ)^l δ_{t+l}

```python
import torch

def compute_gae(rewards, values, dones, gamma=0.99, lam=0.95):
    """
    计算 GAE advantage 和 returns
    Args:
        rewards: [batch_size, seq_len]   每步的 reward
        values:  [batch_size, seq_len+1] Critic 输出的 value（多了最后一步 V(s_{T+1})）
        dones:   [batch_size, seq_len]   是否终止（1=终止）
        gamma:   折扣因子
        lam:     GAE 的 λ 参数
    Returns:
        advantages: [batch_size, seq_len] GAE advantage
        returns:    [batch_size, seq_len] = advantages + values[:, :-1]
    """
    batch_size, seq_len = rewards.shape
    advantages = torch.zeros_like(rewards)
    last_gae_lam = 0.0

    # 反向遍历计算 GAE（递推依赖后面的 advantage）
    for t in reversed(range(seq_len)):
        # 如果当前步是终止状态，下一个 state 的 value 为 0
        mask = 1.0 - dones[:, t].float()

        # TD error = r_t + γ V(s_{t+1}) - V(s_t)
        delta = rewards[:, t] + gamma * values[:, t+1] * mask - values[:, t]

        # A_t = δ_t + γλ A_{t+1}
        last_gae_lam = delta + gamma * lam * mask * last_gae_lam
        advantages[:, t] = last_gae_lam

    # Returns = Value + Advantage
    returns = advantages + values[:, :-1]
    return advantages, returns
```

**面试要点**：
- λ=0 → one-step TD，λ=1 → MC；0<λ<1 在偏差和方差之间平滑调节
- 为什么要反向遍历？GAE 递推依赖后面时刻的 advantage
- dones 的作用？终止状态后没有后续价值，next_value=0

### PPO Value Loss

**核心思想**：和 Policy 类似，Value 也用 clip 防止更新幅度过大（Critic loss 也使用了 clip，防止 value 更新过快影响 advantage 计算的稳定性）。

```python
import torch

def compute_value_loss(new_values, old_values, returns,
                       clip_range=0.2, padding_mask=None):
    """
    计算 PPO 的 value loss（带裁剪）
    Args:
        new_values: [batch_size, seq_len] 当前 critic 的 V(s)
        old_values: [batch_size, seq_len] 收集数据时的 V(s)
        returns:    [batch_size, seq_len] GAE + old_values
        clip_range: value 的裁剪范围
        padding_mask: [batch_size, seq_len] True 表示需要忽略
    """
    # 裁剪 value 到 [old - ε, old + ε]
    v_clipped = old_values + torch.clamp(
        new_values - old_values,
        -clip_range,
        clip_range
    )

    # 计算两种 MSE loss
    v_loss_unclipped = (new_values - returns) ** 2
    v_loss_clipped   = (v_clipped - returns) ** 2

    # 取最大值（保守策略）
    v_loss_max = torch.max(v_loss_unclipped, v_loss_clipped)

    if padding_mask is not None:
        v_loss = 0.5 * v_loss_max.masked_fill(padding_mask, 0.0).sum() / (~padding_mask).sum()
    else:
        v_loss = 0.5 * v_loss_max.mean()

    return v_loss
```

### PPO Policy Loss

**核心公式**：L^CLIP = E[ min(r_t A_t, clip(r_t, 1-ε, 1+ε) A_t) ]，其中 r_t = π_new / π_old

```python
import torch

def compute_policy_loss(log_probs_new, log_probs_old, advantages,
                        clip_range=0.2, padding_mask=None):
    """
    计算 PPO 的 policy loss
    Args:
        log_probs_new: [batch_size, seq_len] 当前策略的 log_probs
        log_probs_old: [batch_size, seq_len] 收集数据时的 log_probs
        advantages:    [batch_size, seq_len] GAE advantage
        clip_range:    裁剪范围 ε
        padding_mask:  [batch_size, seq_len] True 表示需要忽略
    Returns:
        pg_loss:       标量
        clip_fraction: 被 clip 的 token 比例（监控用）
    """
    # 重要性采样比率
    log_ratio = log_probs_new - log_probs_old
    ratio = torch.exp(log_ratio)

    # Advantage 归一化（训练稳定性技巧）
    if padding_mask is not None:
        adv_flat = advantages[~padding_mask]
        advantages = (advantages - adv_flat.mean()) / (adv_flat.std() + 1e-8)
    else:
        advantages = (advantages - advantages.mean()) / (advantages.std() + 1e-8)

    # 计算两种 loss
    pg_loss_1 = -advantages * ratio
    pg_loss_2 = -advantages * torch.clamp(ratio, 1.0 - clip_range, 1.0 + clip_range)

    # 取最大值（保守策略——clip 前后哪个 loss 大用哪个）
    pg_loss = torch.max(pg_loss_1, pg_loss_2)

    if padding_mask is not None:
        pg_loss = pg_loss.masked_fill(padding_mask, 0.0).sum() / (~padding_mask).sum()
    else:
        pg_loss = pg_loss.mean()

    # 计算 clip 比例（用于监控）
    clip_fraction = ((ratio < 1 - clip_range) | (ratio > 1 + clip_range)).float().mean().item()

    return pg_loss, clip_fraction
```

**面试要点**：
- 为什么取 max？实现"悲观"更新——clip 前后哪个 loss 大用哪个，防止激进更新
- Advantage 归一化为什么有用？消除不同 batch 间 advantage 尺度差异的影响

### PPO KL Reward

**核心思想**：奖励中加入 KL 惩罚项，防止策略偏离参考模型太远。

```python
import torch

def compute_kl_and_rewards(ref_log_probs, log_probs, rm_scores,
                           kl_coef=0.1, kl_type="k3"):
    """
    计算 KL 散度和最终的 reward
    Args:
        ref_log_probs: [batch_size, seq_len] 参考模型的 log_prob
        log_probs:     [batch_size, seq_len] 当前策略的 log_prob
        rm_scores:     [batch_size]          Reward Model 给出的序列级得分
        kl_coef:       β，KL 惩罚系数
        kl_type:       "k1" 或 "k3"
    Returns:
        kl:      [batch_size, seq_len] 逐 token 的 KL 散度
        rewards: [batch_size, seq_len] = -β·KL，最后 token 位置 + RM score
    """
    log_r = ref_log_probs - log_probs  # log(ref / current)

    # 两种 KL 估计量
    if kl_type == "k1":
        # D_KL(π || π_ref) = E_π [-log_r]
        kl = -log_r
    elif kl_type == "k3":
        # D_KL(π || π_ref) = E_π [exp(-log_r) - 1 + log_r]
        kl = (log_r.exp() - 1) - log_r
    else:
        raise ValueError(f"Unknown KL estimator: {kl_type}")

    # KL 散度的负值作为奖励（越接近 ref_model 越好）
    # non_score_reward = -β KL：越偏离基准模型，奖励越低
    rewards = -kl_coef * kl

    # 将 RM 得分加到最后一个有效 token 上
    batch_size, seq_len = rewards.shape
    rewards[torch.arange(batch_size), seq_len - 1] += rm_scores

    return kl, rewards
```

**面试要点**：KL 在 RLHF 中的作用是防止策略漂移；k3 无偏且方差小，是 GRPO 常用选择；β 调优——KL 飙升时增大 β，Reward 停滞时适当减小。

### PPO 训练流程伪代码

```python
# PPO 训练主循环
for iteration in range(num_iterations):
    # ========== 1. 采样阶段 ==========
    # 用当前策略（旧策略）采样轨迹
    trajectories = actor.generate(prompts)  # rollout

    # ========== 2. 计算 reward 和 advantage ==========
    # Reward Model 打分
    rm_rewards = reward_model(trajectories)

    # 计算 KL 惩罚
    kl_values = compute_kl(actor, ref_model, trajectories)
    total_rewards = rm_rewards - beta * kl_values  # R_total = R_RM - β·KL

    # 用 Critic 估计每个 token 的状态价值
    values = critic(trajectories)

    # 计算 GAE advantage（从后往前遍历）
    advantages, returns = compute_gae(total_rewards, values, dones, gamma, lam)

    # ========== 3. 多 epoch 更新（复用同一批数据）==========
    for epoch in range(ppo_epochs):
        for mini_batch in data_loader(trajectories):
            # 计算新策略的 log_prob
            new_log_probs = actor(mini_batch)
            ratio = torch.exp(new_log_probs - old_log_probs)

            # Policy Loss（Clip 版本，取 max 保守更新）
            pg_loss = -torch.min(
                ratio * advantages,
                torch.clamp(ratio, 1-eps, 1+eps) * advantages
            ).mean()

            # Value Loss（也带 Clip）
            v_loss = compute_value_loss(critic(mini_batch), old_values, returns)

            # 总 loss + 反向传播更新
            loss = pg_loss + vf_coef * v_loss
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

    # ========== 4. 同步 ref model（如需要）==========
    ref_model.load_state_dict(actor.state_dict())
```
